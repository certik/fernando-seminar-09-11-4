    \begin{python}
  >>> from sympy import *
  >>> x = Symbol('x')

  >>> limit(sin(pi*x)/x, x, 0)
  pi

  >>> integrate(x + sinh(x), x)
  (1/2)*x**2 + cosh(x)

  >>> diff(_, x)
  x + sinh(x)
